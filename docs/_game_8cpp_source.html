169int Game::play_fight() {
170
171 Enemy& enemy1 = enemy_list[enemies_amount++] = Enemy(); // Создаём нового врага и добавляем в конец списка
172
174 Pawn* pl1 = &enemy1, * pl2 = &player, * temp;
175 bool attacking_to_left = true;
176 if (player.get_agility() >= enemy1.get_agility()) { // Меняем местами врага и игрока, если враг не имеет высшую ловкость
177 pl1 = &player;
178 pl2 = &enemy1;
179 attacking_to_left = false;
180 }
181
182 // Отрисовка
183 game.add_enemy(enemy1);
184 game.change_attack_direction(attacking_to_left);
185 game.redraw_console();
186
187 // Ходы
188 int move_number = 0;
189 while (pl1->get_health() > 0 && pl2->get_health() > 0) {
190 pl2->attacked_by(*pl1, move_number);
191 game.redraw_console();
192 if (pl1->get_health() == 0 || pl2->get_health() == 0) break; // Если один умер, завершаем бой
193 // Иначе меняем местами игрока и врага (в переменных)
194 temp = pl1;
195 pl1 = pl2;
196 pl2 = temp;
197 // А также перерисовываем экран
198 attacking_to_left = !attacking_to_left;
199 game.change_attack_direction(attacking_to_left);
200 game.redraw_console();
201 move_number += 1;
202 }
203 if (player.get_health() == 0) { // Если умер игрок
204 return 1;
205 }
206 return 0; // Иначе умер враг
207}
