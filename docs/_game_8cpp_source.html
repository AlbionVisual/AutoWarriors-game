158int Game::play_fight() {
159
160 Enemy& enemy1 = enemy_list[enemies_amount++] = Enemy(); // Создаём нового врага и добавляем в конец списка
161
163 Pawn* pl1 = &enemy1, * pl2 = &player, * temp;
164 bool attacking_to_left = true;
165 if (player.get_agility() >= enemy1.get_agility()) { // Меняем местами врага и игрока, если враг не имеет высшую ловкость
166 pl1 = &player;
167 pl2 = &enemy1;
168 attacking_to_left = false;
169 }
170
171 // Отрисовка
172 game.add_enemy(enemy1);
173 game.change_attack_direction(attacking_to_left);
174 game.redraw_console();
175
176 // Ходы
177 int move_number = 0;
178 while (pl1->get_health() > 0 && pl2->get_health() > 0) {
179 pl2->attacked_by(*pl1, move_number);
180 game.redraw_console();
181 if (pl1->get_health() == 0 || pl2->get_health() == 0) break; // Если один умер, завершаем бой
182 // Иначе меняем местами игрока и врага (в переменных)
183 temp = pl1;
184 pl1 = pl2;
185 pl2 = temp;
186 // А также перерисовываем экран
187 attacking_to_left = !attacking_to_left;
188 game.change_attack_direction(attacking_to_left);
189 game.redraw_console();
190 move_number += 1;
191 }
192 if (player.get_health() == 0) { // Если умер игрок
193 return 1;
194 }
195 return 0; // Иначе умер враг
196}
