19void clear_console() {
20 HANDLE hConsole = GetStdHandle(STD_OUTPUT_HANDLE);
21 COORD coord = { 0, 0 };
22 CONSOLE_SCREEN_BUFFER_INFO csbi;
23 GetConsoleScreenBufferInfo(hConsole, &csbi);
24 DWORD dwBufferSize = csbi.dwSize.X * csbi.dwSize.Y;
25 DWORD dwCharsWritten;
26 FillConsoleOutputCharacter(hConsole, ' ', dwBufferSize, coord, &dwCharsWritten);
27 FillConsoleOutputAttribute(hConsole, 0x1 | 0x2 | 0x4, dwBufferSize, coord, &dwCharsWritten);
28 SetConsoleCursorPosition(hConsole, coord);
29}
