29pawn_specifications Player::get_extra_specifications(const Pawn& target, int move_number) {
30 pawn_specifications extra_specifications = pawn_specifications(0, 0, 0);

32 for (auto& pl_class : TYPES_VALUES) {

69void Player::apply_damage(const Pawn& damager, int extra_damage, int move_number) {
70 int health_change = 0;
71 // Нужно обновлять здоровье в любом случае, поэтому если ничего не применилось вызывается no_posteffect(const Pawn&, const Pawn&, int, int, printer_type) в конце
72 bool post_effect_applied = false;
73
74 for (auto& pl_class : TYPES_VALUES) {
75 int lvl = player_classes.at(pl_class);
76 const vector<PostEffects>& all_effects = POSTEFFECT_RELATION.at(pl_class);
77 for (int i = 0; i < lvl; i++) {
78 const PostEffects& posteffect = all_effects.at(i);
79 if (posteffect == NO_POST_EFFECT) continue;
80 post_effect_applied = true;
81 health_change += get_posteffect(posteffect)(*this, damager, extra_damage, move_number, external_printer);
82 }
83 }
84
85 if (!post_effect_applied) health_change += get_posteffect(NO_POST_EFFECT)(*this, damager, extra_damage, move_number, external_printer);
86
87 if (health_change > 0) health -= health_change;
88 if (health < 0) health = 0;
89}
