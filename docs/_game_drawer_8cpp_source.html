104void GameDrawer::redraw_console() {
105 clear_console();
106 // Если есть какие-то новые данные, эта переменная их оттрекает и заставит окно ещё раз
107 // перерисоваться, чтобы убрать выделение до следующего кадра. Когда вызывается
108 // перерисовка следующего кадра, эта переменная будет всегда false и третьей перерисовки
109 // не будет
110 bool selection_enabled = false;
111 cout << "\tAutoWarriors game\n"; // Начало информационной части консоли
112 cout << "Entity\tClass\t\tHealth\tStrength\tAgility\tEndurance\tWeapon: damage\n";
113 for (short i = enemies_amount; i >= 0; i--) { // Обработка статистики для каждой сущности
114 Pawn* pawn = player_state;
115 Pawn* cashed_pawn = &cashed_player_state;
116 if (i == enemies_amount) { // Первым выводим игрока
117 pawn = player_state;
118 }
119 else {
120 cashed_pawn = &cashed_enemies_state[i];
121 if (i == enemies_amount - 1) { // Затем активного врага
122 pawn = active_enemy_state;
123 }
124 else { // А потом всех остальных
125 pawn = &cashed_enemies_state[i];
126 }
127 }
128 string pawn_type = (i == enemies_amount) ? "Player" : "Enemy";
129 string visible_name = (i == enemies_amount) ? player_state->get_visible_name() : (i == enemies_amount - 1) ? active_enemy_state->get_visible_name() : cashed_enemies_state[i].get_visible_name();
130 cout << pawn_type + "\t" + visible_name + "\t"; // Начало вывода статистики одной сущности
131 if (visible_name.length() < 8) cout << "\t";
132 selection_enabled = colored_print(pawn->health, cashed_pawn->health) | selection_enabled;
133 cout << "\t";
134 selection_enabled = colored_print(pawn->specifications.strength, cashed_pawn->specifications.strength) | selection_enabled;
135 cout << "\t\t";
136 selection_enabled = colored_print(pawn->specifications.agility, cashed_pawn->specifications.agility) | selection_enabled;
137 cout << "\t";
138 selection_enabled = colored_print(pawn->specifications.endurance, cashed_pawn->specifications.endurance) | selection_enabled;
139 cout << "\t\t" + pawn->active_weapon.weapon_name + ": ";
140 selection_enabled = colored_print(pawn->active_weapon.damage, cashed_pawn->active_weapon.damage) | selection_enabled;
141 cout << "\n"; // Конец вывода статистики одной сущности
142 }
143 cout << "\n";
144 if (enemies_amount > 0) { // Если враги есть, то рисуем бой
145 string direction_arrow = attack_left ? "<--" : "-->";
146 cout << "\t\t";
147 if (player_state->health < cashed_player_state.health) { // Если есть изменения в здоровье игрока, то отображаем это слева
148 colorize_text(RED_WHITE);
149 cout << "-" + to_string(cashed_player_state.health - player_state->health);
150 colorize_text();
151 selection_enabled = true; // Тоже просим перерисовать консоль, чтобы через секунду это выделение пропало
152 }
153 cout << "\tYou\t" + direction_arrow + "\t" + active_enemy_state->get_visible_name() + "\t";
154 int& last_enemy_health = active_enemy_state->health;
155 int& last_cashed_enemy_health = cashed_enemies_state[enemies_amount - 1].health;
156 if (last_enemy_health < last_cashed_enemy_health) { // Если есть изменения здоровья врага, то тоже отрисовываем это рядом с врагом
157 colorize_text(GREEN_WHITE);
158 cout << "-" + to_string(last_cashed_enemy_health - last_enemy_health);
159 colorize_text();
160 selection_enabled = true;
161 }
162 cout << "\n\n";
163 } // Конец отрисовки боя
164
165 // Отрисовка всех активных сообщений (они не учавствуют в автоматической отмене
166 // перерисовки, т.к. имеют собсвтенный таймер, поэтому selection_enabled не устанавливается
167 // в true, но изменяется оставшееся время)
168 for (auto message_iter = message_list.begin(); message_iter != message_list.end(); message_iter++) {
169 Message& message = *message_iter;
170 if (message.seconds <= 0) { // Проверка времени (удаляем, если время истекло)
171 auto temp = message_iter--;
172 message_list.erase(temp);
173 continue;
174 };
175 colorize_text(message.color); // Отрисовываем окрашенный текст
176 cout << message.text;
177 colorize_text();
178 message.seconds -= 1; // Уменьшаем оставшееся время
179 }
180
181
182 if (selection_enabled) { // Если был запрос на отмену временного выделения
183 sync_cache();
184 freeze(); // то вызываем перерисовку после небольшой паузы
185 redraw_console();
186 }
187 else freeze(); // Или просто ждём
188}
